```python
    """Validate file path is safe — no traversal, no null bytes, within project."""
    if "\x00" in file_path:
        raise FileToolsError(f"Invalid file path: contains null bytes: {file_path!r}")
    resolved = Path(file_path).resolve()
    if ".." in Path(file_path).parts:

    if project_root:
        root = Path(project_root).resolve()
        if not str(resolved).startswith(str(root)):

    return resolved
```

```python
#!/usr/bin/env python3
"""#315: Scan for hardcoded secrets in source files."""
import re
import sys
from pathlib import Path

SECRET_PATTERNS = [
    (r'(?i)(api[_-]?key|secret|token|password)\s*=\s*["\'][^"\']{8,}["\']', "Potential hardcoded secret"),
    (r'(?i)bearer\s+[a-zA-Z0-9\-._~+/]+=*', "Bearer token"),
    (r'(?:AKIA|AGPA|AIDA|AROA|AIPA|ANPA|ANVA|ASIA)[A-Z0-9]{16}', "AWS access key"),
    (r'sk-[a-zA-Z0-9]{32,}', "OpenAI API key pattern"),
    (r'ghp_[a-zA-Z0-9]{36}', "GitHub personal access token"),
]

IGNORE_PATHS = {".git", "node_modules", "__pycache__", ".venv", "memory/brain.db"}
IGNORE_FILES = {".env.example", "pytest.ini", "requirements.txt"}

def scan_file(path: Path) -> list[tuple[int, str, str]]:
    findings = []
    try:
        text = path.read_text(encoding="utf-8", errors="ignore")
    except (PermissionError, OSError):
        return findings
    for i, line in enumerate(text.splitlines(), 1):
        if line.strip().startswith("#") or "PLACEHOLDER" in line:
            continue
        for pattern, desc in SECRET_PATTERNS:
            if re.search(pattern, line):
                findings.append((i, desc, line.strip()[:80]))
    return findings

def main():
    root = Path(sys.argv[1]) if len(sys.argv) > 1 else Path(".")
    total = 0
    for path in sorted(root.rglob("*")):
        if any(part in IGNORE_PATHS for part in path.parts):
            continue
        if path.name in IGNORE_FILES or not path.is_file():
            continue
        if path.suffix in {".py", ".json", ".yml", ".yaml", ".toml", ".cfg", ".sh", ".env"}:
            findings = scan_file(path)
            for line_no, desc, snippet in findings:
                print(f"{path}:{line_no}: {desc}: {snippet}")
                total += 1
    if total:
        print(f"\n{total} potential secret(s) found")
        sys.exit(1)
    print("No hardcoded secrets found")

if __name__ == "__main__":
    main()
```

```python
# core/failure_router.py
"""Failure routing logic extracted from orchestrator (#317)."""
from enum import Enum, auto

class FailureAction(Enum):
    RETRY_ACT = auto()
    REPLAN = auto()
    SKIP = auto()
    ABORT = auto()

ENVIRONMENTAL_KEYWORDS = ["network", "connection", "timeout", "dns", "certificate"]

class FailureRouter:
    def __init__(self, max_act_retries: int = 3):
        self.max_act_retries = max_act_retries

    def route_failure(
        self, phase: str, attempt: int, error: str,
        verification_output: dict | None = None
    ) -> FailureAction:
        error_lower = error.lower() if error else ""
        if any(kw in error_lower for kw in ENVIRONMENTAL_KEYWORDS):
            return FailureAction.SKIP
        if phase == "sandbox":
            return FailureAction.RETRY_ACT if attempt < self.max_act_retries else FailureAction.REPLAN
        if phase == "verify":
            if verification_output and verification_output.get("tests_passed", 0) > 0:
                return FailureAction.RETRY_ACT if attempt < self.max_act_retries else FailureAction.REPLAN
            return FailureAction.REPLAN
        return FailureAction.SKIP
```

```dockerfile
# Dockerfile — optimized multi-stage build
FROM python:3.12-slim AS builder

RUN apt-get update && apt-get install -y --no-install-recommends \
    git ca-certificates && \
    rm -rf /var/lib/apt/lists/*

WORKDIR /build
COPY requirements.txt .
RUN pip install --no-cache-dir --prefix=/install -r requirements.txt

COPY . .

# Runtime stage — minimal image
FROM python:3.12-slim

# Security: non-root user
RUN groupadd -r aura && useradd -r -g aura -d /app -s /sbin/nologin aura

COPY --from=builder /install /usr/local
COPY --from=builder /build /app

WORKDIR /app
RUN chown -R aura:aura /app

USER aura

ENV PYTHONUNBUFFERED=1 \
    PYTHONDONTWRITEBYTECODE=1 \
    AURA_SKIP_CHDIR=1

VOLUME ["/app/memory", "/app/logs"]
EXPOSE 8000

HEALTHCHECK --interval=30s --timeout=5s --start-period=10s --retries=3 \
    CMD python3 -c "import urllib.request; urllib.request.urlopen('http://localhost:8000/health')" || exit 1

CMD ["uvicorn", "tools.mcp_server:app", "--host", "0.0.0.0", "--port", "8000"]
```

- `--no-install-recommends` reduces apt package size
- `--no-cache-dir` on pip install
- `PYTHONDONTWRITEBYTECODE=1` avoids .pyc files
- Replace curl healthcheck with Python (no curl in slim image)
- Add `--start-period=10s` for startup grace period
